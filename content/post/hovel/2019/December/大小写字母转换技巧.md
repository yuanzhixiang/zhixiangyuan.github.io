```java
    /**
     * 小写转大写
     *
     * @param c 小写字母
     * @return 大写字母
     */
    public static char upper(char c) {
        return (char) (c & 0xdf);
    }

    /**
     * 大写转小写
     *
     * @param c 大写字母
     * @return 小写字母
     */
    public static char lower(char c) {
        return (char) (c | 0x20);
    }

    /**
     * 大小写字母互转
     */
    public static char reverse(char c) {
        return (char) (c ^ 0x20);
    }
```
